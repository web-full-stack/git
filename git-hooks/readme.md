# git hooks

```sh
$ cd ./.git && ls -al

```

```sh

$ code ./pre-commit.sample
# OR
$ cat ./pre-commit.sample

$ vim ./pre-commit.sample
# ESC + :q, exit vim
# ESC + :!q, exit vim
$ vi ./pre-commit.sample
$ nano ./pre-commit.sample
# Control + X, exit nano

```
